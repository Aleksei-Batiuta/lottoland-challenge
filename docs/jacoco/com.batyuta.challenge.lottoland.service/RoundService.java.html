package com.batyuta.challenge.lottoland.service;

import com.batyuta.challenge.lottoland.dao.RoundDAO;
import com.batyuta.challenge.lottoland.enums.StatusEnum;
import com.batyuta.challenge.lottoland.model.RoundEntity;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.Collection;

/**
 * Round DAO Service.
 */
@Service
public class RoundService {
    /**
     * Round DAO.
     */
    private final RoundDAO roundDAO;

    /**
     * Default Constructor.
     *
     * @param dao round DAO
     */
    @Autowired
    public RoundService(final RoundDAO dao) {
        this.roundDAO = dao;
    }

    /**
     * Getter for all rounds.
     *
     * @return round
     * @see RoundDAO#getAll()
     */
    public Collection<RoundEntity> getAllRounds() {
        return roundDAO.getAll();
    }

    /**
     * Saves round.
     *
     * @param round round
     * @return round
     * @see RoundDAO#create(RoundEntity)
     */
    public RoundEntity save(final RoundEntity round) {
        return roundDAO.create(round);
    }

    /**
     * Gets rounds by user ID.
     *
     * @param userId user ID
     * @return rounds
     */
    public Collection<RoundEntity> getRoundsByUserId(
            final int userId) {
        return roundDAO.getRoundsByUserId(userId);
    }

    /**
     * Deletes all rounds by user ID.
     *
     * @param userId user ID
     */
    public void deleteAllRoundsByUserId(final int userId) {
        roundDAO.deleteAllByUserId(userId);
    }

    /**
     * Gets total played rounds.
     *
     * @return number of total played rounds
     */
    public int getTotalRounds() {
        return roundDAO.getTotalRounds();
    }

    /**
     * Gets total played rounds by status.
     *
     * @param status status of rounds
     * @return number of total played rounds
     */
    public int getTotalRounds(final StatusEnum status) {
        return roundDAO.getTotalRoundsByStatus(status);
    }
}
